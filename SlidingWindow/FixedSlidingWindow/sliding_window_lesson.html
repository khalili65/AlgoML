def max_sum_subarray(arr, k):
    # 1. First Window
    current_sum = 0
    for i in range(k):
        current_sum += arr[i]
    
    # 2. Store Best
    max_sum = current_sum

    # 3. Iterate Rest
    for i in range(k, len(arr)):
        
        # 4. THE SLIDE
        entering = arr[i]
        leaving  = arr[i - k]
        
        current_sum = current_sum + entering - leaving

        # 5. Update Best
        max_sum = max(max_sum, current_sum)

    return max_sum
